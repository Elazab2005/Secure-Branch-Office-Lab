! ============================================
! IOU3 - Access Switch
! Hostname: SW-IOU3
! Management IP: 192.168.30.3/24
! Connected Devices: PC3 (Staff), PC4 (Guest)
! ============================================

enable
configure terminal
hostname SW-IOU3

! Create VLANs
vlan 10
 name Staff
vlan 20
 name Guest
vlan 30
 name Management
exit

! EtherChannel Po2 to MLS (IOU1)
interface range e0/0-1
 channel-group 2 mode active
 channel-protocol lacp
exit

interface port-channel 2
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30
exit

! PC3 - Staff (VLAN 10)
interface e0/2
 switchport mode access
 switchport access vlan 10
exit

! PC4 - Guest (VLAN 20)
interface e0/3
 switchport mode access
 switchport access vlan 20
exit

! Management SVI
interface vlan 30
 ip address 192.168.30.3 255.255.255.0
 no shutdown
exit

! Default Gateway
ip default-gateway 192.168.30.1

! SSH Configuration
ip domain-name clinic.local
crypto key generate rsa modulus 2048
username Soliman password 1234
ip ssh version 2

line vty 0 4
 login local
 transport input ssh
exit

enable secret cisco

! Save configuration
write memory
