! ============================================
! IOU2 - Access Switch
! Hostname: SW-IOU2
! Management IP: 192.168.30.2/24
! Connected Devices: PC1 (Staff), PC2 (Guest)
! ============================================

enable
configure terminal
hostname SW-IOU2

! Create VLANs
vlan 10
 name Staff
vlan 20
 name Guest
vlan 30
 name Management
exit

! EtherChannel Po1 to MLS (IOU1)
interface range e0/0-1
 channel-group 1 mode active
 channel-protocol lacp
exit

interface port-channel 1
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30
exit

! PC1 - Staff (VLAN 10)
interface e0/2
 switchport mode access
 switchport access vlan 10
exit

! PC2 - Guest (VLAN 20)
interface e0/3
 switchport mode access
 switchport access vlan 20
exit

! Management SVI
interface vlan 30
 ip address 192.168.30.2 255.255.255.0
 no shutdown
exit

! Default Gateway
ip default-gateway 192.168.30.1

! SSH Configuration
ip domain-name clinic.local
crypto key generate rsa modulus 2048
username Soliman password 1234
ip ssh version 2

line vty 0 4
 login local
 transport input ssh
exit

enable secret cisco

! Save configuration
write memory
